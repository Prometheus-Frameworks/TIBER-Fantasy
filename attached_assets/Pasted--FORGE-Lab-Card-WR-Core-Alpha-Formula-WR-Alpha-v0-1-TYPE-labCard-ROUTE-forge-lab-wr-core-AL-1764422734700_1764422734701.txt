    { key: "TS_norm" },

    const WinSkill  = CC_norm;

    const WR_Core   = 0.40 * Chain + 0.40 * Explosive + 0.20 * WinSkill;
    const WR_Alpha  = 25 + 65 * WR_Core;

    return {
      TS_norm: round4(TS_norm),

      FD_norm: round4(FD_norm),
      YAC_norm: round4(YAC_norm),
      CC_norm: round4(CC_norm),
      Chain: round4(Chain),
      Explosive: round4(Explosive),
      WinSkill: round4(WinSkill),
      WR_Core: round4(WR_Core),
      WR_Alpha: round2(WR_Alpha)
    };
  },
  ui: {
    sections: [
      { title: "Normalized Metrics", keys: ["TS_norm","YPRR_norm","FD_norm","YAC_norm","CC_norm"]},
      { title: "Subscores", keys: ["Chain","Explosive","WinSkill"]},
