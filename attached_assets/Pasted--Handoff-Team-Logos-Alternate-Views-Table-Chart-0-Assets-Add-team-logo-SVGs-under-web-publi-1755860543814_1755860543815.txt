import React from "react";

type Props = { team: string; size?: number; className?: string };

export const TeamLogo: React.FC<Props> = ({ team, size = 24, className }) => {
  const src = `/logos/${team}.svg`;
  const [err, setErr] = React.useState(false);

  if (err) {
    // Fallback badge (no logo file yet)
    return (
      <div
        className={`inline-flex items-center justify-center rounded-full border border-gray-300 bg-gray-50 text-gray-700 ${className || ""}`}
        style={{ width: size, height: size, fontSize: Math.max(10, Math.floor(size * 0.45)) }}
        title={team}
      >
        {team}
      </div>
    );
  }

  return (
    <img
      src={src}
      alt={`${team} logo`}
      width={size}
      height={size}
      className={`inline-block ${className || ""}`}
      onError={() => setErr(true)}
      title={team}
    />
  );
};

3) Utility: tier â†’ color

File: web/src/lib/sosColors.ts

export function tierColor(score: number) {
  if (score >= 67) return "bg-green-100 text-green-900 border-green-300";
  if (score >= 33) return "bg-yellow-100 text-yellow-900 border-yellow-300";
  return "bg-red-100 text-red-900 border-red-300";
}


Use this both in the table and the chart to keep colors consistent.

File: web/src/components/SOSBarChart.tsx

import React from "react";
import {
  ResponsiveContainer, BarChart, Bar, XAxis, YAxis, Tooltip, CartesianGrid, LabelList
} from "recharts";
import { tierColor } from "../lib/sosColors";
import { TeamLogo } from "./TeamLogo";

type Item = {
  team: string;
  opponent: string;
  score: number; // final score 0-100
};

type Props = { items: Item[]; limit?: number };

export const SOSBarChart: React.FC<Props> = ({ items, limit = 16 }) => {
  const data = [...items]
    .sort((a, b) => b.score - a.score)
    .slice(0, limit)
    .map((d) => ({
      ...d,
      colorClass: tierColor(d.score),
    }));

  // Inline color mapping via CSS classes is tricky in Recharts, so weâ€™ll use a single series
  // and style the bars via a function fallback on fill (CSS variables).
  return (
    <div className="w-full h-[520px]">
      <ResponsiveContainer>
        <BarChart data={data} margin={{ top: 20, right: 20, bottom: 40, left: 20 }}>
          <CartesianGrid strokeDasharray="3 3" />
          <XAxis dataKey="team" />
          <YAxis domain={[0, 100]} />
          <Tooltip
            formatter={(val: any, _name, entry: any) => [`${val}`, `Score`]}
            labelFormatter={(team) => {
              const row = data.find((d) => d.team === team);
              return row ? `${row.team} vs ${row.opponent}` : `${team}`;
            }}
          />
          <Bar dataKey="score" isAnimationActive>
            <LabelList dataKey="score" position="top" />
          </Bar>
        </BarChart>
      </ResponsiveContainer>

      {/* Legend with logos (top N) */}
      <div className="grid grid-cols-4 gap-2 mt-3">
        {data.map((d) => (
          <div key={d.team} className={`flex items-center gap-2 px-2 py-1 rounded border ${d.colorClass}`}>
            <TeamLogo team={d.team} size={18} />
            <div className="text-sm">{d.team} vs {d.opponent}</div>
            <div className="ml-auto font-semibold">{Math.round(d.score)}</div>
          </div>
        ))}
      </div>
    </div>
  );
};


Note: Recharts bars are a single color by default. If you want true per-bar color, we can switch to a Cell map later. For now, the tier color is reinforced in the legend chips.
