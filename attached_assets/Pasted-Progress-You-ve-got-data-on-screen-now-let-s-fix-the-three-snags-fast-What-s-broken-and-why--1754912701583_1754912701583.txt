3) Make search actually return “Waddle”

Use the real search endpoint when q.length >= 2, and normalize comparisons. Also show client‑side results immediately while the network returns.

const norm = (s='') => s.normalize().toLowerCase().trim();

useEffect(() => {
  const id = setTimeout(async () => {
    const qn = norm(q);
    if (qn.length < 2) return; // keep rankings view
    // optimistic local filter
    const local = baseRows.filter(p =>
      [p.name, p.team, p.id].some(v => norm(v||'').includes(qn))
    );
    setRows(local);
    // network search
    const res = await api.redraftPlayers({ search: q, pos: "WR", limit: 50 });
    setRows((res.data || []).map(mapPlayerLite));
  }, 250);
  return () => clearTimeout(id);
}, [q, baseRows]);

(Where baseRows is the rankings you loaded on mount.)
