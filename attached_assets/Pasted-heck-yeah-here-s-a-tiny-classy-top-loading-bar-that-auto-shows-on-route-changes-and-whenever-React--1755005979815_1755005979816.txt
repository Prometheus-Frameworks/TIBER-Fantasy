heck yeah—here’s a tiny, classy top loading bar that auto‑shows on route changes and whenever React Query is fetching. Gold→purple gradient, minimal code.

1) Install

npm i nprogress

2) Add the bar styles

Create src/styles/nprogress.css:

/* Thin top bar */
#nprogress { pointer-events:none; }
#nprogress .bar {
  background: linear-gradient(90deg,#F5C043,#8B5CF6);
  position: fixed; z-index: 9999; top: 0; left: 0;
  width: 100%; height: 3px; box-shadow: 0 0 8px rgba(245,192,67,.6);
}
/* Optional subtle spinner on the right */
#nprogress .peg {
  display: block; position: absolute; right: 0; width: 100px; height: 100%;
  box-shadow: 0 0 10px #8B5CF6, 0 0 5px #F5C043; opacity: .8; transform: rotate(3deg) translate(0,-4px);
}
/* Fade out */
#nprogress .bar,
#nprogress .peg { transition: all 180ms ease; }

3) Global Progress hook

Create src/ui/useTopProgress.tsx:

import { useEffect } from "react";
import NProgress from "nprogress";
import "nprogress/nprogress.css"; // base
import "../styles/nprogress.css";  // our skin
import { useIsFetching } from "@tanstack/react-query";
import { useNavigation } from "react-router-dom";

NProgress.configure({ showSpinner: false, trickleSpeed: 120, minimum: 0.08 });

export function useTopProgress() {
  const isFetching = useIsFetching();              // any React Query in-flight?
  const navigation = useNavigation();              // route transitions (RR v6.4+)
  const navBusy = navigation.state === "loading" || navigation.state === "submitting";
  const busy = navBusy || isFetching > 0;

  useEffect(() => {
    // small delay avoids flicker on super-fast ops
    let t: number | null = null;
    if (busy) {
      t = window.setTimeout(() => NProgress.start(), 100);
    } else {
      if (t) window.clearTimeout(t);
      NProgress.done(true);
    }
    return () => { if (t) window.clearTimeout(t); };
  }, [busy]);
}

4) Mount it once at the app root

In your App.tsx (or RootLayout.tsx) near the top of the component:

import { useTopProgress } from "./ui/useTopProgress";

export default function App() {
  useTopProgress();  // <— one line, now it “just works”
  // ...your routes/layout
}

5) (Optional) Also show on manual fetches outside React Query

If you have any raw fetch()/axios calls, you can wrap them:

// fetchWithProgress.ts
import NProgress from "nprogress";
export async function fetchWithProgress(input: RequestInfo, init?: RequestInit) {
  NProgress.start();
  try { return await fetch(input, init); }
  finally { NProgress.done(true); }
}
