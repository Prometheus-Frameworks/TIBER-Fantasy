```typescript
// Load all 100 player baseline chunks
const playerChunks = [
  ...qbChunks,  // 10 QB chunks from earlier
  ...rbChunks,  // 34 RB chunks from Grok
  ...wrChunks,  // 36 WR chunks from Grok
  ...teChunks   // 20 TE chunks from Grok

for (const chunk of playerChunks) {
  // 1. Generate embedding
  const embedding = await generateEmbedding(chunk.content);

  await db.query(`

    VALUES ($1, $2, $3, NOW())
  `, [
    chunk.content,
    `[${embedding.join(',')}]`,
    JSON.stringify(chunk.metadata)
  ]);
}
```

## Phase 2: NFLverse Snap Count Integration (Future)

**After 2024 baseline data is stable, integrate live usage data:**

### Data Source
`https://github.com/nflverse/nflverse-data/releases/download/snap_counts/snap_counts_2024.csv`

### Integration Steps

1. **Download and parse CSV**
```python
import pandas as pd

snap_data = pd.read_csv('https://github.com/nflverse/nflverse-data/releases/download/snap_counts/snap_counts_2024.csv')

# Aggregate by player
player_snaps = snap_data.groupby(['player', 'position', 'team']).agg({
    'offense_snaps': 'sum',
    'offense_pct': 'mean'
}).reset_index()
```

2. **Create database table**
```sql
CREATE TABLE snap_counts_2024 (
  id SERIAL PRIMARY KEY,
  player_name VARCHAR(255),
  position VARCHAR(10),
  team VARCHAR(10),
  total_snaps INTEGER,
  avg_snap_pct DECIMAL(5,2),
  season INTEGER DEFAULT 2024,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_snap_counts_player ON snap_counts_2024(player_name);
CREATE INDEX idx_snap_counts_position ON snap_counts_2024(position);
```

3. **Update jargon mapping**
```json
{
  "snap_share": {
    "nflfastr_field": "offense_pct",
    "calculation": "AVG(offense_pct) WHERE player_name = ?",
    "description": "Average % of team offensive snaps",
    "queryable": true,
    "data_source": "nflverse snap_counts_2024.csv",
    "note": "2024 season data available for training/comparison"
  }
}
```

4. **Query endpoint**
```typescript
app.get('/api/snap-counts-2024/:playerName', async (req, res) => {
  const { playerName } = req.params;
  
  const result = await db.query(`
    SELECT player_name, position, team, total_snaps, avg_snap_pct
    FROM snap_counts_2024
    WHERE LOWER(player_name) LIKE LOWER($1)
  `, [`%${playerName}%`]);
  
  return res.json({
    available: result.rows.length > 0,
    season: 2024,
    data: result.rows,
    note: "2024 baseline data - for comparison only, not current season"
  });
});
```
