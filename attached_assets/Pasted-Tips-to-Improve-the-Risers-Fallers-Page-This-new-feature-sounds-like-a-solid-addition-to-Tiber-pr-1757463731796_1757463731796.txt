Sample Python Code for Parsing
Here’s a basic script using requests and pandas to fetch, parse, and identify risers/fallers. Assume your API returns JSON like {"players": [{"name": "Travis Etienne", "position": "RB", "power_score": 85, "prev_score": 70, "notes": "143 yards, 8.9 YPC"}]}. Run this post-games to update Tiber’s dataset.
import requests
import pandas as pd

# Hypothetical API endpoint and key
API_URL = "https://tiber.com/api/power-rankings/week/1"
API_KEY = "your_api_key_here"

# Fetch data
response = requests.get(API_URL, headers={"Authorization": f"Bearer {API_KEY}"})
if response.status_code != 200:
    raise Exception(f"API error: {response.status_code}")

data = response.json()["players"]

# Parse into DataFrame
df = pd.DataFrame(data)
df["delta_score"] = df["power_score"] - df["prev_score"]

# Identify risers, fallers, steady
risers = df[df["delta_score"] > 5].sort_values("delta_score", ascending=False)
fallers = df[df["delta_score"] < -5].sort_values("delta_score")
steady = df[(df["delta_score"] >= -5) & (df["delta_score"] <= 5)]

# Example: Add performance-based logic (e.g., for RBs)
rb_risers = risers[(risers["position"] == "RB") & (risers["ypc"] > 5.0)]  # Assuming 'ypc' field

# Output or save to DB (e.g., to CSV for simplicity)
risers.to_csv("week1_risers.csv", index=False)
fallers.to_csv("week1_fallers.csv", index=False)
steady.to_csv("week1_steady.csv", index=False)

print("Risers:", risers[["name", "delta_score", "notes"]])
print("Fallers:", fallers[["name", "delta_score", "notes"]])
